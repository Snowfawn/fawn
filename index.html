#!/bin/bash echo "1을 눌러 설치를 진행하세요. q를 눌러 종료하세요." read -n 1 choice if [ "$choice" == "1" ]; then # 필요한 디렉토리 생성 FAWN_DIR="$HOME/.snowfawn/fawn" mkdir -p "$HOME/.snowfawn" # git clone 수행 if [ ! -d "$FAWN_DIR" ]; then git clone https://github.com/Snowfawn/fawn.git "$FAWN_DIR" else echo "이미 fawn이 설치되어 있습니다." exit 1 fi # 심볼릭 링크 생성 sudo ln -s "$FAWN_DIR/fawn.sh" /usr/local/bin/fawn # 실행 권한 부여 chmod +x "$FAWN_DIR/fawn.sh" echo "설정이 완료되었습니다. 이제 'fawn' 명령어로 스크립트를 실행할 수 있습니다." elif [ "$choice" == "q" ]; then echo "종료합니다." exit 0 else echo "잘못된 입력입니다. 종료합니다." exit 1 fi
